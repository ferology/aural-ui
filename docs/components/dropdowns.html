    </div>

<script>
// Dropdown functionality is initialized by Aural.initDropdowns()
// Or manually handle toggle:
const trigger = document.querySelector('.dropdown-trigger');
const menu = document.querySelector('.dropdown-menu');

trigger.addEventListener('click', () => {
    const expanded = trigger.getAttribute('aria-expanded') === 'true';
    trigger.setAttribute('aria-expanded', !expanded);
    menu.hidden = expanded;
});
</script>
const Dropdown = ({ label, items, variant = 'primary' }) => {
  const [isOpen, setIsOpen] = useState(false);
  const dropdownRef = useRef(null);

  useEffect(() => {
    const handleClickOutside = (e) => {
      if (dropdownRef.current && !dropdownRef.current.contains(e.target)) {
        setIsOpen(false);
      }
    };
    document.addEventListener('click', handleClickOutside);
    return () => document.removeEventListener('click', handleClickOutside);
  }, []);

  return (
    <div className="dropdown" ref={dropdownRef}>
      <button
        className={`btn btn-${variant} dropdown-trigger`}
        aria-expanded={isOpen}
        aria-haspopup="true"
        onClick={() => setIsOpen(!isOpen)}
      >
        {label}
        <i data-lucide="chevron-down" />
      </button>
      <div className="dropdown-menu" role="menu" hidden={!isOpen}>
        {items.map((item, index) => (
          item.divider ? (
            <div key={index} className="dropdown-divider" role="separator" />
          ) : (
            <a
              key={index}
              href={item.href || '#'}
              className={`dropdown-item ${item.danger ? 'dropdown-item-danger' : ''}`}
              role="menuitem"
              onClick={() => {
                item.onClick?.();
                setIsOpen(false);
              }}
            >
              {item.icon && <i data-lucide={item.icon} />}
              {item.label}
            </a>
          )
        ))}
      </div>
    </div>
  );
};

// Usage
<Dropdown
  label="Options"
  items={[
    { label: 'Profile', icon: 'user', onClick: () => console.log('Profile') },
    { label: 'Settings', icon: 'settings' },
    { divider: true },
    { label: 'Sign Out', icon: 'log-out', danger: true }
  ]}
/>
<template>
  <div class="dropdown" ref="dropdown">
    <button
      class="btn dropdown-trigger"
      :class="`btn-${variant}`"
      :aria-expanded="isOpen"
      aria-haspopup="true"
      @click="toggle"
    >
      {{ label }}
      <i data-lucide="chevron-down"></i>
    </button>
    <div class="dropdown-menu" role="menu" :hidden="!isOpen">
      <template v-for="(item, index) in items" :key="index">
        <div v-if="item.divider" class="dropdown-divider" role="separator"></div>
        <a
          v-else
          href="#"
          class="dropdown-item"
          :class="{ 'dropdown-item-danger': item.danger }"
          role="menuitem"
          @click.prevent="handleItemClick(item)"
        >
          <i v-if="item.icon" :data-lucide="item.icon"></i>
          {{ item.label }}
        </a>
      </template>
    </div>
  </div>
</template>

<script>
export default {
  props: {
    label: { type: String, required: true },
    items: { type: Array, required: true },
    variant: { type: String, default: 'primary' }
  },
  data() {
    return { isOpen: false };
  },
  methods: {
    toggle() {
      this.isOpen = !this.isOpen;
    },
    handleItemClick(item) {
      item.onClick?.();
      this.isOpen = false;
    }
  }
}
</script>

Best Practices

When to Use

  • Action Menus: Provide multiple actions for an item (edit, share, delete)
  • User Menus: Account settings, profile, sign out options
  • Bulk Actions: Actions on selected table rows or list items
  • View Options: Filter, sort, or display options for content
  • Navigation Overflow: Secondary navigation items

Organization

  • Logical Grouping: Group related actions with dividers
  • Frequent First: Place most common actions at the top
  • Destructive Last: Place delete/remove actions at the bottom
  • Color for Danger: Use .dropdown-item-danger for destructive actions

Accessibility

  • Use semantic <button> elements for triggers
  • Add aria-expanded to show open/closed state
  • Add aria-haspopup="true" to indicate dropdown behavior
  • Support keyboard: Enter/Space to open, Arrow keys to navigate, Esc to close
  • Use aria-label on icon-only buttons
  • Each dropdown must have a unique id attribute

Required Structure

<div class="dropdown" id="unique-id">
  <button class="dropdown-trigger" aria-expanded="false" aria-haspopup="true">
    Trigger Text
  </button>
  <div class="dropdown-menu" role="menu" hidden>
    <a href="#" class="dropdown-item" role="menuitem">Item</a>
  </div>
</div>

Available Classes

  • .dropdown - Container element (required)
  • .dropdown-trigger - Button that opens the menu
  • .dropdown-menu - The menu container
  • .dropdown-item - Individual menu items
  • .dropdown-divider - Horizontal separator
  • .dropdown-header - Section header text
  • .dropdown-item-danger - Destructive action styling
  • .dropdown-right - Right-align the menu
