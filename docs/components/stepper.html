const Stepper = ({ steps, currentStep }) => {
  const getStepState = (index) => {
    if (index < currentStep) return 'completed';
    if (index === currentStep) return 'active';
    return 'pending';
  };

  return (
    <div className="aural-stepper">
      {steps.map((step, index) => {
        const state = getStepState(index);
        return (
          <div
            key={index}
            className={`aural-step ${state !== 'pending' ? `aural-step--${state}` : ''}`}
          >
            <div
              className="aural-step__indicator"
              aria-current={state === 'active' ? 'step' : undefined}
            >
              {state === 'completed' ? (
                <i data-lucide="check" />
              ) : (
                index + 1
              )}
            </div>
            {index < steps.length - 1 && (
              <div className="aural-step__connector" />
            )}
            <div className="aural-step__content">
              <div className="aural-step__title">{step.title}</div>
              <div className="aural-step__description">{step.description}</div>
            </div>
          </div>
        );
      })}
    </div>
  );
};

// Usage
const steps = [
  { title: 'Account', description: 'Create your account' },
  { title: 'Profile', description: 'Add your details' },
  { title: 'Complete', description: 'Review and finish' }
];

<Stepper steps={steps} currentStep={1} />
<template>
  <div class="aural-stepper">
    <div
      v-for="(step, index) in steps"
      :key="index"
      :class="stepClasses(index)"
    >
      <div
        class="aural-step__indicator"
        :aria-current="index === currentStep ? 'step' : undefined"
      >
        <i v-if="index < currentStep" data-lucide="check"></i>
        <template v-else>{{ index + 1 }}</template>
      </div>
      <div v-if="index < steps.length - 1" class="aural-step__connector"></div>
      <div class="aural-step__content">
        <div class="aural-step__title">{{ step.title }}</div>
        <div class="aural-step__description">{{ step.description }}</div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  props: {
    steps: { type: Array, required: true },
    currentStep: { type: Number, default: 0 }
  },
  methods: {
    stepClasses(index) {
      const classes = ['aural-step'];
      if (index < this.currentStep) classes.push('aural-step--completed');
      if (index === this.currentStep) classes.push('aural-step--active');
      return classes.join(' ');
    }
  }
}
</script>

Best Practices

When to Use

  • Use steppers for multi-step processes (forms, checkout, onboarding)
  • Show users their progress through a sequential workflow
  • Use horizontal steppers for 3-5 steps on desktop
  • Use vertical steppers for more than 5 steps or mobile layouts
  • Ideal for tracking order status or delivery progress

Step States

  • .aural-step--completed - Step has been finished (use check icon)
  • .aural-step--active - Current step (add aria-current="step")
  • .aural-step--error - Step has an error (use X icon)
  • Default (no class) - Upcoming/pending step

Content Guidelines

  • Keep step titles concise (1-2 words)
  • Use descriptions to provide additional context
  • Number steps when order matters (1, 2, 3, 4)
  • Use icons when they add clarity (checkmark for complete, error icon for failed)
  • Keep the total number of steps between 3-7 for best usability

Interactive vs Non-Interactive

  • Non-Interactive: Show progress only (order tracking)
  • Interactive: Allow clicking to jump between steps (form wizard)
  • For interactive steppers, make completed steps clickable
  • Disable or hide future steps that cannot be accessed yet
  • Consider validation before allowing users to proceed

Horizontal vs Vertical

  • Horizontal: Better for 3-5 steps on desktop, shows all steps at once
  • Vertical: Better for 5+ steps or mobile, allows for longer descriptions
  • Vertical steppers work better with timestamps or detailed status updates
  • Horizontal steppers are more compact and save vertical space

Accessibility

  • Include role="navigation" on the stepper container
  • Add aria-label to describe the stepper (e.g., "Checkout progress")
  • Use aria-current="step" on the active step indicator
  • Add aria-label to each step indicator describing its state
  • Use semantic icons (check for complete, X for error)
  • Ensure color is not the only indicator of step state
  • Provide text alternatives for icon-only indicators

Mobile Considerations

  • Consider using vertical steppers on mobile for better readability
  • Ensure step indicators are at least 4040px for touch targets
  • Hide step descriptions on very small screens if needed
  • Allow horizontal scrolling for many horizontal steps on mobile

Form Wizard Pattern

  • Place stepper at the top of the form
  • Provide "Back" and "Next" buttons below form content
  • Validate current step before allowing users to proceed
  • Save progress automatically or provide a "Save Draft" option
  • Show a summary/review step before final submission
