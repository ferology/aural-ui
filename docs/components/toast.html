Toast

Temporary notification messages that appear at the corner of the screen.

Basic Toasts
View Code 
// Show a success toast
Aural.showToast('Operation completed successfully!', 'success');

// Show an error toast
Aural.showToast('Error: Failed to save changes.', 'error');
All Types
View Code 
Aural.showToast('Success message', 'success');
Aural.showToast('Informational message', 'info');
Aural.showToast('Warning message', 'warning');
Aural.showToast('Error message', 'error');
Toast with Title
View Code 
// Third parameter is the title
Aural.showToast('Your profile has been updated.', 'success', 'Success');
Aural.showToast('Unable to connect to server.', 'error', 'Connection Error');
Custom Duration
View Code 
// Fourth parameter is duration in milliseconds
Aural.showToast('Quick message', 'info', null, 2000);  // 2 seconds
Aural.showToast('Extended message', 'info', null, 8000);  // 8 seconds
Multiple Toasts
View Code 
// Toasts stack automatically
Aural.showToast('First notification', 'success');
setTimeout(() => Aural.showToast('Second notification', 'info'), 500);
setTimeout(() => Aural.showToast('Third notification', 'warning'), 1000);

Multiple toasts automatically stack in the corner, with newer toasts appearing below existing ones.

Common Use Cases

Best Practices

When to Use

  • Brief feedback for user actions (save, delete, copy)
  • Status updates that don't require user action
  • Background process completions
  • Non-critical notifications

Toast Types

  • success - Successful operations
  • info - Informational updates
  • warning - Warnings that don't block user
  • error - Errors and failures

JavaScript API

  • Aural.showToast(message, type) - Basic toast
  • Aural.showToast(message, type, title) - With title
  • Aural.showToast(message, type, title, duration) - Custom duration

Duration Guidelines

  • Default: 4000ms (4 seconds)
  • Short messages: 2000-3000ms
  • Important messages: 6000-8000ms
  • Keep duration long enough to read

Accessibility

  • Toasts are announced to screen readers
  • Don't put critical actions in toasts
  • Ensure sufficient display time
  • Use clear, concise messages

<script>
function showToast(message = 'This is a toast message', variant = 'info') {
    const container = document.getElementById('toast-container');

    const toast = document.createElement('div');
    toast.className = `toast toast-${variant}`;

    const icons = {
        info: 'info',
        success: 'check-circle',
        warning: 'alert-triangle',
        error: 'x-circle'
    };

    toast.innerHTML = `
        <div class="toast-icon">
            <i data-lucide="${icons[variant]}"></i>
        </div>
        <div class="toast-content">
            <p class="toast-message">${message}</p>
        </div>
        <button class="toast-close" aria-label="Close toast">
            <i data-lucide="x"></i>
        </button>
    `;

    container.appendChild(toast);
    lucide.createIcons();

    // Auto dismiss after 3 seconds
    setTimeout(() => {
        toast.classList.add('toast-hiding');
        setTimeout(() => toast.remove(), 300);
    }, 3000);

    // Close button handler
    toast.querySelector('.toast-close').addEventListener('click', () => {
        toast.classList.add('toast-hiding');
        setTimeout(() => toast.remove(), 300);
    });
}
</script>
import { useState, useEffect } from 'react';

const Toast = ({ message, variant = 'info', duration = 3000, onClose }) => {
  const [isVisible, setIsVisible] = useState(true);

  const icons = {
    info: 'info',
    success: 'check-circle',
    warning: 'alert-triangle',
    error: 'x-circle'
  };

  useEffect(() => {
    if (duration) {
      const timer = setTimeout(() => {
        setIsVisible(false);
        setTimeout(() => onClose(), 300);
      }, duration);
      return () => clearTimeout(timer);
    }
  }, [duration, onClose]);

  const handleClose = () => {
    setIsVisible(false);
    setTimeout(() => onClose(), 300);
  };

  return (
    <div className={`toast toast-${variant} ${!isVisible ? 'toast-hiding' : ''}`}>
      <div className="toast-icon">
        <i data-lucide={icons[variant]}></i>
      </div>
      <div className="toast-content">
        <p className="toast-message">{message}</p>
      </div>
      <button className="toast-close" onClick={handleClose} aria-label="Close toast">
        <i data-lucide="x"></i>
      </button>
    </div>
  );
};

// Usage
function App() {
  const [toasts, setToasts] = useState([]);

  const showToast = (message, variant = 'info') => {
    const id = Date.now();
    setToasts(prev => [...prev, { id, message, variant }]);
  };

  const removeToast = (id) => {
    setToasts(prev => prev.filter(toast => toast.id !== id));
  };

  return (
    <>
      <button
        className="btn btn-primary"
        onClick={() => showToast('This is a toast message', 'success')}
      >
        Show Toast
      </button>

      <div className="toast-container">
        {toasts.map(toast => (
          <Toast
            key={toast.id}
            message={toast.message}
            variant={toast.variant}
            onClose={() => removeToast(toast.id)}
          />
        ))}
      </div>
    </>
  );
}
<template>
  <div>
    <button class="btn btn-primary" @click="showToast">
      Show Toast
    </button>

    <div class="toast-container">
      <transition-group name="toast">
        <div
          v-for="toast in toasts"
          :key="toast.id"
          :class="`toast toast-${toast.variant}`"
        >
          <div class="toast-icon">
            <i :data-lucide="getIcon(toast.variant)"></i>
          </div>
          <div class="toast-content">
            <p class="toast-message">{{ toast.message }}</p>
          </div>
          <button
            class="toast-close"
            @click="removeToast(toast.id)"
            aria-label="Close toast"
          >
            <i data-lucide="x"></i>
          </button>
        </div>
      </transition-group>
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      toasts: []
    };
  },
  methods: {
    showToast(message = 'This is a toast message', variant = 'info', duration = 3000) {
      const id = Date.now();
      this.toasts.push({ id, message, variant });

      if (duration) {
        setTimeout(() => this.removeToast(id), duration);
      }
    },
    removeToast(id) {
      this.toasts = this.toasts.filter(toast => toast.id !== id);
    },
    getIcon(variant) {
      const icons = {
        info: 'info',
        success: 'check-circle',
        warning: 'alert-triangle',
        error: 'x-circle'
      };
      return icons[variant] || 'info';
    }
  }
};
</script>

    showToast = true;

</script>
