When to Use

  • File Browsers: Display hierarchical file and folder structures
  • Navigation Menus: Create nested navigation with expandable sections
  • Category Organization: Show nested categories or taxonomies
  • Settings Panels: Organize grouped settings hierarchically
  • Organizational Charts: Display hierarchical relationships

Accessibility

  • Use role="tree" on the container and role="treeitem" on items
  • Implement aria-expanded to indicate expand/collapse state
  • Support arrow key navigation (up/down to move, left/right to collapse/expand)
  • Use aria-selected to indicate the currently focused item
  • Provide aria-label for toggle buttons
  • Ensure visible focus indicators on all interactive elements
  • Use proper heading hierarchy when tree items have content

Best Practices

Visual Design

  • Use clear indentation to show hierarchy levels
  • Use connecting lines or indentation guides for deep nesting
  • Differentiate folders from files with distinct icons
  • Show expand/collapse icons only on items with children
  • Use badges to show item counts or status

Interaction

  • Click on toggle icon or entire row to expand/collapse
  • Double-click or Enter to select/activate an item
  • Support multi-select with Shift/Ctrl clicks when needed
  • Provide context menus for additional actions
  • Remember expand/collapse state across sessions

Performance

  • Lazy load children for large trees
  • Virtualize rendering for very large datasets
  • Show loading indicators while fetching children
  • Consider initial collapse state for large hierarchies

Content

  • Keep labels concise but descriptive
  • Use consistent icon styles throughout
  • Group related items together
  • Consider alphabetical or logical ordering

Common Mistakes to Avoid

  • Not providing keyboard navigation support
  • Missing visual indication of current selection
  • Inconsistent indentation or hierarchy visualization
  • Loading entire tree upfront for large datasets
  • No visual feedback during expand/collapse transitions
  • Forgetting to handle empty states for folders
  • Not differentiating between expandable and leaf items

Framework Examples

class TreeView {
  constructor(data, containerId) {
    this.data = data;
    this.container = document.getElementById(containerId);
    this.expanded = new Set();
    this.selected = null;
    this.render();
    this.setupEventListeners();
  }

  setupEventListeners() {
    this.container.addEventListener('click', (e) => {
      const button = e.target.closest('.aural-tree__content');
      if (!button) return;

      const item = button.closest('.aural-tree__item');
      const itemId = item.dataset.id;

      // Handle toggle for expandable items
      if (button.querySelector('.aural-tree__toggle i')) {
        e.stopPropagation();
        this.toggleNode(itemId);
      } else {
        // Handle selection for all items
        this.selectNode(itemId);
      }
    });

    // Keyboard navigation
    this.container.addEventListener('keydown', (e) => {
      const item = e.target.closest('.aural-tree__item');
      if (!item) return;

      const itemId = item.dataset.id;

      if (e.key === 'ArrowRight') {
        e.preventDefault();
        this.expandNode(itemId);
      } else if (e.key === 'ArrowLeft') {
        e.preventDefault();
        this.collapseNode(itemId);
      } else if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        this.selectNode(itemId);
      }
    });
  }

  toggleNode(id) {
    if (this.expanded.has(id)) {
      this.collapseNode(id);
    } else {
      this.expandNode(id);
    }
  }

  expandNode(id) {
    this.expanded.add(id);
    this.render();
  }

  collapseNode(id) {
    this.expanded.delete(id);
    this.render();
  }

  selectNode(id) {
    this.selected = id;
    this.render();

    // Trigger callback if provided
    const node = this.findNode(this.data, id);
    if (node && node.onSelect) {
      node.onSelect(node);
    }
  }

  findNode(nodes, id) {
    for (const node of nodes) {
      if (node.id === id) return node;
      if (node.children) {
        const found = this.findNode(node.children, id);
        if (found) return found;
      }
    }
    return null;
  }

  renderNode(node, level = 0) {
    const isExpanded = this.expanded.has(node.id);
    const isSelected = this.selected === node.id;
    const hasChildren = node.children && node.children.length > 0;

    return `
      <li
        class="aural-tree__item ${hasChildren ? (isExpanded ? '' : 'aural-tree__item--collapsed') : 'aural-tree__item--leaf'}"
        data-id="${node.id}"
      >
        <button
          class="aural-tree__content ${isSelected ? 'is-selected' : ''}"
          tabindex="0"
        >
          <span class="aural-tree__toggle">
            ${hasChildren ? `<i data-lucide="${isExpanded ? 'chevron-down' : 'chevron-right'}"></i>` : ''}
          </span>
          <span class="aural-tree__icon ${node.iconColor ? `aural-tree__icon--${node.iconColor}` : ''}">
            <i data-lucide="${node.icon}"></i>
          </span>
          <span class="aural-tree__label">${node.label}</span>
          ${node.badge ? `<span class="aural-tree__badge">${node.badge}</span>` : ''}
        </button>
        ${hasChildren && isExpanded ? `
          <ul class="aural-tree__children">
            ${node.children.map(child => this.renderNode(child, level + 1)).join('')}
          </ul>
        ` : ''}
      </li>
    `;
  }

  render() {
    this.container.innerHTML = `
      <ul class="aural-tree">
        ${this.data.map(node => this.renderNode(node)).join('')}
      </ul>
    `;

    // Re-initialize Lucide icons
    if (window.lucide) lucide.createIcons();
  }
}

const TreeNode = ({ node, level = 0, expanded, selected, onToggle, onSelect }) => {
  const isExpanded = expanded.has(node.id);
  const isSelected = selected === node.id;
  const hasChildren = node.children && node.children.length > 0;

  const handleToggle = (e) => {
    e.stopPropagation();
    onToggle(node.id);
  };

  const handleSelect = () => {
    onSelect(node.id, node);
  };

  return (
    <li
      className={`aural-tree__item ${
        hasChildren
          ? isExpanded
            ? ''
            : 'aural-tree__item--collapsed'
          : 'aural-tree__item--leaf'
      }`}
    >
      <button
        className={`aural-tree__content ${isSelected ? 'is-selected' : ''}`}
        onClick={handleSelect}
        tabIndex={0}
      >
        <span className="aural-tree__toggle" onClick={handleToggle}>
          {hasChildren && (
            isExpanded ? <ChevronDown size={16} /> : <ChevronRight size={16} />
          )}
        </span>
        <span
          className={`aural-tree__icon ${
            node.iconColor ? `aural-tree__icon--${node.iconColor}` : ''
          }`}
        >
          {node.icon}
        </span>
        <span className="aural-tree__label">{node.label}</span>
        {node.badge && (
          <span className="aural-tree__badge">{node.badge}</span>
        )}
      </button>
      {hasChildren && isExpanded && (
        <ul className="aural-tree__children">
          {node.children.map((child) => (
            <TreeNode
              key={child.id}
              node={child}
              level={level + 1}
              expanded={expanded}
              selected={selected}
              onToggle={onToggle}
              onSelect={onSelect}
            />
          ))}
        </ul>
      )}
    </li>
  );
};

const TreeView = ({ data, onNodeSelect }) => {
  const [expanded, setExpanded] = useState(new Set());
  const [selected, setSelected] = useState(null);

  const handleToggle = (id) => {
    setExpanded((prev) => {
      const next = new Set(prev);
      if (next.has(id)) {
        next.delete(id);
      } else {
        next.add(id);
      }
      return next;
    });
  };

  const handleSelect = (id, node) => {
    setSelected(id);
    if (onNodeSelect) {
      onNodeSelect(node);
    }
  };

  return (
    <ul className="aural-tree">
      {data.map((node) => (
        <TreeNode
          key={node.id}
          node={node}
          expanded={expanded}
          selected={selected}
          onToggle={handleToggle}
          onSelect={handleSelect}
        />
      ))}
    </ul>
  );
};

<script>
import { ref } from 'vue';
import TreeNode from './TreeNode.vue';

export default {
  components: { TreeNode },
  props: {
    data: { type: Array, required: true }
  },
  emits: ['node-select'],
  setup(props, { emit }) {
    const expanded = ref(new Set());
    const selected = ref(null);

    const handleToggle = (id) => {
      if (expanded.value.has(id)) {
        expanded.value.delete(id);
      } else {
        expanded.value.add(id);
      }
      // Trigger reactivity
      expanded.value = new Set(expanded.value);
    };

    const handleSelect = (id, node) => {
      selected.value = id;
      emit('node-select', node);
    };

    return {
      expanded,
      selected,
      handleToggle,
      handleSelect
    };
  }
};
</script>

import { computed } from 'vue';
import { ChevronRight, ChevronDown } from 'lucide-vue-next';

export default {
  name: 'TreeNode',
  components: { ChevronRight, ChevronDown },
  props: {
    node: { type: Object, required: true },
    expanded: { type: Set, required: true },
    selected: { type: String, default: null }
  },
  emits: ['toggle', 'select'],
  setup(props) {
    const isExpanded = computed(() => props.expanded.has(props.node.id));
    const isSelected = computed(() => props.selected === props.node.id);

    );

    return { isExpanded, isSelected, hasChildren };
  }
};

<script>

  import { ChevronRight, ChevronDown } from 'lucide-svelte';

  export let node;
  export let expanded;
  export let selected;
  export let level = 0;

  $: isExpanded = $expanded.has(node.id);
  $: isSelected = $selected === node.id;
  $: hasChildren = node.children && node.children.length > 0;

    e.stopPropagation();
    dispatch('toggle', node.id);
  }

  function handleSelect() {
    dispatch('select', { id: node.id, node });
  }
</script>
